collections

collections.namedtuple

              
import collections

Point = collections.namedtuple("Point", ["x", "y"])
p = Point(3, 4)

import math

with namedtuple we can have more semantic tuples and avoid using mysterious indexes to access tuple's members

collections.defaultdict

              
import collections

s = [('submit', 'create_instance'),
     ('delete', 'notify'),
     ('submit', 'send_email'),
     ('touch', 'scroll'),]
d = collections.defaultdict(list)
for k, v in s:
    d[k].append(v)

sorted(d.items())
# [('delete', ['notify']),
#  ('submit', ['create_instance', 'send_email']),
#  ('touch', ['scroll'])]
defaultdict subclasses dict, where you can provide initial value

Functional programming modules

itertools, functools, operator
functional programming gems in stdlib, which are often forgot or unkown

itertools

Infinite: count, cycle, repeat

Sequence: accumulate, chain, filterfalse, groupby, ...

Combinatorics: permutations, product, combinations, ...
itertools has a set of fast and memory efficient functions, which return iterators

itertools.cycle

              
import itertools

c = itertools.cycle('ABC')
next(c)
# 'A'
next(c)
# 'B'
next(c)
# 'C'
next(c)
# 'A'

itertools.filterfalse

              
import itertools

# leave uneven numbers
ff = itertools.filterfalse(lambda x: x % 2 == 0, range(10))
for i in ff:
    print(i)
# 1
# 3
# 5
# 7
# 9

itertools.groupby

              
from datetime import datetime as dt
import itertools

# work start and finish log
log = [(dt(2010, 1, 1, 8), 'arrive'),
       (dt(2010, 1, 1, 16), 'leave'),
       (dt(2010, 1, 2, 9), 'arrive'),
       (dt(2010, 1, 2, 16), 'leave')]

g =  itertools.groupby(sorted(log, key=lambda x: x[0].hour),
                       key=lambda x: x[0].hour):
for key, group in g:
    print(key, list(group))
# 8 [(datetime.datetime(2010, 1, 1, 8, 0), 'arrive')]
# 9 [(datetime.datetime(2010, 1, 2, 9, 0), 'arrive')]
# 16 [(datetime.datetime(2010, 1, 1, 16, 0), 'leave'),
#     (datetime.datetime(2010, 1, 2, 16, 0), 'leave')]

itertools.product

              
import itertools

# ('b', 1)

wsgiref

from wsgiref.simple_server import make_server

def hello_world_app(environ, start_response):
    status = '200 OK'  # HTTP Status
    headers = [('Content-type', 'text/plain; charset=utf-8')]
    start_response(status, headers)

    # The returned object is going to be printed
    return [b"Hello World"]

with make_server('', 8000, hello_world_app) as httpd:
    print("Serving on port 8000...")

    # Serve until process is killed
    httpd.serve_forever()
